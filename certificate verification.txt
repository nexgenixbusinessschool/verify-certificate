        document.getElementById("verifyForm").addEventListener("submit", function(event) {
            event.preventDefault();

            // Get the entered certificate ID
            const certificateID = document.getElementById("certificateID").value;

            // Certificate data with ID, owner's name, and course details
            const certificates = [
                { id: "NGBS001-28012025-CC01", owner: "Gioia Antonette Pavan", course: "Lesson 1 The Basics" },
                { id: "NGBS001-28012025-AC01", owner: "Gioia Antonette Pavan", course: "From Branding to Sales Mastery", averageScore: 98.88 },
                { id: "NGBS001-28012025-CO01", owner: "Gioia Antonette Pavan", course: "From Branding to Sales Mastery" },
                { id: "NGBS001-28012025-CC02", owner: "Jayvee Brylle Limjoco", course: "Lesson 1 The Basics" },
                { id: "NGBS001-28012025-AC02", owner: "Jayvee Brylle Limjoco", course: "From Branding to Sales Mastery", averageScore: 98.88 },
                { id: "NGBS001-28012025-CO02", owner: "Jayvee Brylle Limjoco", course: "From Branding to Sales Mastery" },
              	{ id: "NGBS007-28012025-AC02", owner: "Windsor Gene Ancheta", course: "Leadership and Organizational Behavior", averageScore: 94.34 },
                { id: "NGBS000-19032025-AC00", owner: "Jane Doe", course: "Sample Course", averageScore: 99.99 },
                { id: "NGBS000-19032025-CC00", owner: "Jane Doe", course: "Sample Course"},
                { id: "NGBS000-19032025-CO00", owner: "Jane Doe", course: "Sample Course"},
            ];

            const resultDiv = document.getElementById("verificationResult");
            const ownerDiv = document.getElementById("certificateOwner");
            const courseDiv = document.getElementById("courseDetails");
            
            // Check if the certificate ID is valid
            const certificate = certificates.find(cert => cert.id === certificateID);

            if (certificate) {
                // If valid, show certificate owner and success message
                resultDiv.textContent = "Certificate is valid!";
                resultDiv.style.color = "green";
                ownerDiv.textContent = `Certificate Owner: ${certificate.owner}`;
                ownerDiv.style.color = "green";
                
                // Display course details for all certificate types
                if (certificateID.includes("AC")) {
                    courseDiv.textContent = `Course Name: ${certificate.course}, Average Score: ${certificate.averageScore}`;
                } else if (certificateID.includes("CC") || certificateID.includes("CO")) {
                    courseDiv.textContent = `Course Name: ${certificate.course}`;
                } else {
                    courseDiv.textContent = ""; // Clear course details if not relevant
                }
                courseDiv.style.color = "blue";
            } else {
                // If invalid, show error message
                resultDiv.textContent = "Invalid certificate!";
                resultDiv.style.color = "red";
                ownerDiv.textContent = ""; // Clear the owner information
                courseDiv.textContent = ""; // Clear course details
            }
        });
    </script>
